model = LogisticRegression(max_iter=10000, class_weight="balanced", random_state=22)
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.33, random_state=22)
model.fit(X_train.values, y_train)
pred = model.predict(X_test)

def show_performance(title, pred_):
    labels = [
        "Accuracy",
        "Recall",
        "Precision"
    ]
    values = [
        accuracy_score(pred_, y_test),
        recall_score(pred_, y_test),
        precision_score(pred_, y_test)
    ]
    plt.barh(labels, values)
    plt.title(title)
    plt.show()
 
def _construct_selected_observations():

explainer = LimeTabularExplainer(
    training_data=X.values,  
    feature_names=X.columns,
    mode="classification"
)

def decompose_with_lime():
    for i in range(selected_observations_X.shape[0]):
        observation = selected_observations_X.iloc[[i]]
        explanation = explainer.explain_instance(
            data_row=observation.iloc[0],
            predict_fn=lambda d: model.predict_proba(d)
        )   
        _ = explanation.as_pyplot_figure()
        
def check_stability_of_lime_explanations():
